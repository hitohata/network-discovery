6use crate::commands::DiscoveryCommand;
7use crate::store::data_store::DataStore;
8use tracing::{Level, error};
9
10const BROADCAST_ADDRESS: &str = "255.255.255.255";
11
12#[tokio::main]
13async fn main() {
14    // set up tracing for logging
15    tracing_subscriber::fmt()
16        .with_env_filter(Level::INFO.as_str())
17        .with_thread_ids(true)
18        .with_thread_names(true)
19        .init();
20
21    // List to hold thread handlers
22    let mut handlers = vec![];
23
24    // channel for commands
25    let (command_tx, command_rx): (
26        tokio::sync::mpsc::Sender<DiscoveryCommand>,
27        tokio::sync::mpsc::Receiver<DiscoveryCommand>,
28    ) = tokio::sync::mpsc::channel(32);
29    // channel for data
30    let (response_tx, response_rx): (
31        tokio::sync::broadcast::Sender<shared::schemas::target_messages::ResponseSchema>,
32        tokio::sync::broadcast::Receiver<shared::schemas::target_messages::ResponseSchema>,
33    ) = tokio::sync::broadcast::channel(32);
34
35    let data_store_command_tx = command_tx.clone();
36    let data_store = std::sync::Arc::new(tokio::sync::Mutex::new(DataStore::new()));
37
38    // Start the TargetServer in a separate thread
39    let target_handler = tokio::spawn(async {
40        let target_server = shared::server::target_server::TargetServer::new();
41        target_server.run().await.unwrap_or_else(|e| {
42            error!("Failed to run TargetServer: {}", e);
43        });
44    });
45    handlers.push(target_handler);
46
47    // Start the DiscoverServer in a separate thread
48    let discovery_server = crate::threads::discovery_server::DiscoveryServer::new();
49    let discover_server_handler =
50        tokio::spawn(async move { discovery_server.run(command_rx, response_tx.clone()).await });
51    handlers.push(discover_server_handler);
52
53    // run data store service
54    let data_store_service =
55        crate::threads::data_store_service::DataStoreService::new(data_store.clone());
56    data_store_service.run(data_store_command_tx, response_rx);
57
58    for handler in handlers {
59        if let Err(e) = handler.await {
60            error!("Thread panicked: {:?}", e);
61        }
62    }
63}
